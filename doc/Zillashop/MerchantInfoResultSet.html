def initialize(hsh)
  @hash = hsh
  if merchant_info_set.is_a? Hash
    @merchants = [MerchantInfoResult.new(merchant_info_set)]
  else
    @merchants = merchant_info_set.map { |m| MerchantInfoResult.new(m) }
  end
end
