def initialize(hsh)
  @hash = hsh
  if brands_set["Brand"].is_a? Hash
    @brands = [BrandsResult.new(brands_set["Brand"])]
  else
    @brands = brands_set["Brand"].map { |p| BrandsResult.new(p) }
  end
end
